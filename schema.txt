CREATE TYPE blood_group_enum AS ENUM ('A+', 'A-', 'B+', 'B-', 'AB+', 'AB-', 'O+', 'O-');
CREATE TYPE blood_component_enum AS ENUM ('WHOLE_BLOOD', 'PLATELETS', 'PLASMA');
CREATE TYPE request_status_enum AS ENUM ('PENDING', 'PARTIALLY_FULFILLED', 'FULFILLED', 'CANCELLED');
CREATE TYPE ping_status_enum AS ENUM ('NOTIFIED', 'ACCEPTED', 'DECLINED', 'EXPIRED');

CREATE TABLE users (
    user_id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    phone VARCHAR(15) UNIQUE NOT NULL,
    password CHAR(60) NOT NULL,
    blood_group blood_group_enum NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE donor_profiles (
    donor_id SERIAL PRIMARY KEY,
    user_id INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    current_location GEOMETRY(Point, 4326) NOT NULL,
    location_updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_donation_date DATE,
    last_donation_type blood_component_enum,
    is_available BOOLEAN DEFAULT TRUE
);
CREATE TABLE blood_requests (
    request_id SERIAL PRIMARY KEY,
    requester_id INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    patient_name VARCHAR(100) NOT NULL,
    hospital_name VARCHAR(100) NOT NULL,
    hospital_location GEOMETRY(Point, 4326) NOT NULL,
    blood_group blood_group_enum NOT NULL,
    required_component blood_component_enum NOT NULL,
    units_required INT NOT NULL CHECK (units_required > 0),
    units_fulfilled INT DEFAULT 0 CHECK (units_fulfilled >= 0),
    secure_token VARCHAR(255) UNIQUE NOT NULL,
    status request_status_enum DEFAULT 'PENDING',
    cancellation_reason TEXT,
    verification_slip_url VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT chk_fulfillment_cap CHECK (units_fulfilled <= units_required)
);
CREATE TABLE blood_requests_archive (
    request_id SERIAL PRIMARY KEY,
    requester_id INT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    patient_name VARCHAR(100) NOT NULL,
    hospital_name VARCHAR(100) NOT NULL,
    hospital_location GEOMETRY(Point, 4326) NOT NULL,
    blood_group blood_group_enum NOT NULL,
    required_component blood_component_enum NOT NULL,
    units_required INT NOT NULL CHECK (units_required > 0),
    units_fulfilled INT DEFAULT 0 CHECK (units_fulfilled >= 0),
    secure_token VARCHAR(255) UNIQUE NOT NULL,
    status request_status_enum DEFAULT 'PENDING',
    cancellation_reason TEXT,
    verification_slip_url VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT chk_fulfillment_cap CHECK (units_fulfilled <= units_required)
);
CREATE TABLE request_pings (
    ping_id SERIAL PRIMARY KEY,
    request_id INT NOT NULL REFERENCES blood_requests(request_id) ON DELETE CASCADE,
    donor_id INT NOT NULL REFERENCES donor_profiles(donor_id) ON DELETE CASCADE,
    status ping_status_enum DEFAULT 'NOTIFIED',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
